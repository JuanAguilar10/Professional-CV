\newenvironment{indentsection}{
	% Crea una sección identada
	\begin{list}{}{
		\setlength{\leftmargin}{\newparindent}
		\setlength{\parsep}{0pt}
		\setlength{\parskip}{0pt}
		\setlength{\itemsep}{0pt}
		\setlength{\topsep}{0pt}}
	}
	{\end{list}}

\newenvironment{personaltabledata}{
	% Tabla de datos personales
	\spacedhrule{1.0em}{0em}
	\roottitle{\nompersonaldata}
	\begingroup
	\setlength{\tabcolsep}{0.7em}
	\setlength{\fboxsep}{0pt}
	\setlength{\columnsep}{-35pt}
	\def\arraystretch{1.339}
	\begin{wrapfigure}{l}{0.24\textwidth}
		\vspace{-0.33cm}
		\fbox{\centering\includegraphics[width=3.63cm]{\fotoautor}}
	\end{wrapfigure}
	
	\noindent
	\vspace{-0.79cm}
	\begin{table}[H]
		\centering
		\hspace*{-1.3cm}
		\begin{tabular}{ll}
		}
		{
		\end{tabular}
	\end{table}
	\vspace{-0.3cm}
	\endgroup
}

\newenvironment{summary}{
	% Párrafo de resúmen
	\spacedhrule{1.0em}{0em}
	\roottitle{\nomsummary}
	\begingroup
	\vspace{-1.0em}
	\begin{multicols}{2}
	\noindent}
	{
	\end{multicols}
	\vspace{-0.1cm}
	\endgroup
}

\newenvironment{cvblock}[1]{
	% Crea un bloque de contenido
	%	#1	Título
	\spacedhrule{1.0em}{0em}
	\ifx\hfuzz#1\hfuzz
		\throwwarning{Titulo del bloque no definido}
		\stop
	\else
		\roottitle{#1}
	\fi
	}
	{
}

\newenvironment{signature}{
	% Crea la firma del usuario
	%	#1	Título
	\vfill
	\begin{flushright}
		\begin{tabular}{c}
			\includegraphics[width=5cm]{\firmaautor} \\
		}
		{
		\end{tabular}
	\end{flushright}
}

\newenvironment{institution}[3][]{
	% Nueva institución
	%	#1	Opcional: url o link institución
	%	#2	Nombre institución
	%	#3	Ubicación institución
	\nopagebreak[4]
	\begin{indentsection}
		\item []
		\ifx\hfuzz#1\hfuzz
			\textscale{1.1}{\textcolor{\linkcolor}{#2}}
		\else
			\textscale{1.1}{\href{#1}{#2}}
		\fi
		\hfill \textsc{#3} \break \vspace{-1em}
		}{
	\end{indentsection}
	\nopagebreak[4]
}
