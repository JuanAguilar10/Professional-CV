% Crea una sección identada
\newenvironment{indentsection}{
	\begin{list}{}{
		\setlength{\leftmargin}{\newparindent}
		\setlength{\parsep}{0pt}
		\setlength{\parskip}{0pt}
		\setlength{\itemsep}{0pt}
		\setlength{\topsep}{0pt}}
	}{
	\end{list}
}

% Párrafo de descripción
\newenvironment{summary}{
	\spacedhrule{1.0em}{0em}
	\roottitle{\nomsummary}
	\begingroup
	\vspace{-0.5em}
	\begin{multicols}{2}
	\noindent
	}{
	\end{multicols}
	\vspace{-0.1cm}
	\endgroup
}

% Párrafo de descripción con foto de perfil
\newenvironment{photosummary}{
	\spacedhrule{1.0em}{0em}
	\roottitle{\nomsummary}
	\begingroup
	\setlength{\tabcolsep}{0.7em}
	\setlength{\fboxsep}{0pt}
	\setlength{\columnsep}{-0.3cm}
	\def\arraystretch{1.339}
	\vspace{0.2em}
	\begin{wrapfigure}{l}{0.24\textwidth}
		\vspace{-0.45cm}
		\fbox{\centering\includegraphics[width=\userphotosize cm,height=\userphotosize cm]{\fotoautor}}
	\end{wrapfigure}
	\noindent
	}{
	\endgroup
}

% Tabla de datos personales
\newenvironment{personaltabledata}{
	\spacedhrule{1.0em}{0em}
	\roottitle{\nompersonaldata}
	\begingroup
	\setlength{\tabcolsep}{0.7em}
	\setlength{\fboxsep}{0pt}
	\setlength{\columnsep}{-35pt}
	\def\arraystretch{1.339}
	\begin{wrapfigure}{l}{0.24\textwidth}
		\vspace{-0.33cm}
		\fbox{\centering\includegraphics[width=\userphotosize cm,height=\userphotosize cm]{\fotoautor}}
	\end{wrapfigure}
	\noindent
	\vspace{-0.71cm}
	\begin{table}[H]
		\centering
		\hspace*{-2.1cm}
		\begin{tabular}{ll}
		}{
		\end{tabular}
	\end{table}
	\endgroup
}

% Crea un bloque de contenido
%	#1	Título
\newenvironment{cvblock}[1]{
	\spacedhrule{1.0em}{0em}
	\ifx\hfuzz#1\hfuzz
		\throwwarning{Titulo del bloque no definido}
		\stop
	\else
		\roottitle{#1}
	\fi
	}{
}

% Nueva institución
%	#1	Opcional: url o link institución
%	#2	Nombre institución
%	#3	Ubicación institución
\newenvironment{institution}[3][]{
	\nopagebreak[4]
	\begin{indentsection}
		\item []
		\ifx\hfuzz#1\hfuzz
			\textscale{1.1}{\textcolor{\linkcolor}{#2}}
		\else
			\textscale{1.1}{\href{#1}{#2}}
		\fi
		\hfill \textsc{#3} \break \vspace{-1em}
		}{
	\end{indentsection}
	\nopagebreak[4]
}

% Crea la firma del usuario
%	#1	Título
\newenvironment{signature}{
	\vfill
	\begin{flushright}
		\begin{tabular}{c}
			\includegraphics[width=\usersignaturesize cm]{\firmaautor} \\
		}{
		\end{tabular}
	\end{flushright}
}
